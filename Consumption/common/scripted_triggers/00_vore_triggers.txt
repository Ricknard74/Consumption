can_vore = {
    #Females Only
    trigger_if = {
        limit = { global_var:gender_allowed = 0 }
        is_female = yes
    }
    #Males Only
    trigger_if = {
        limit = { global_var:gender_allowed = 1 }
        is_female = no
    }
    is_alive = yes
    age >= 18
    has_character_flag = vore_capable 
    trigger_if = {
        limit = { has_variable = num_prey }
        var:num_prey < max_belly_capacity
    } 
    NOR = {
        has_character_flag = vored
        has_character_flag = potential_prey
        has_trait = incapable
        has_trait = digesting
    }
    OR = {
        global_var:oral_banned = no
        global_var:anal_banned = no
        global_var:unbirth_banned = no
        global_var:cock_banned = no
    }
}
