# Event: Playful Possession (Milestone 1/2)
# The schemer swallows one of target's artifacts during a disagreement, turning it into flirtation
seduce_vore.1010 = {
    type = character_event
    title = seduce_vore.1010.t
    desc = seduce_vore.1010.desc
    theme = seduce_scheme
    
    left_portrait = {
        character = root
        animation = schadenfreude
    }
    
    right_portrait = {
        character = scope:target
        animation = shock
    }
    
    lower_left_portrait = root
    
    widget = {
        gui = "event_window_widget_scheme"
        container = "custom_widgets_container"
    }
    
    cooldown = { years = 2 }
    
    trigger = {
        NOT = { has_variable = scheme_event_active_flag }
        can_vore = yes
        scope:target = {
            can_be_prey = yes
            any_character_artifact = {
                count >= 1
                is_equipped = no
                is_unique = no
            }
        }
    }
    
    immediate = {
        set_variable = scheme_event_active_flag
        
        scope:target = {
            ordered_character_artifact = {
                limit = {
                    is_unique = no
                }
                order_by = {
                    value = artifact_rarity
                    multiply = -1
                }
                save_scope_as = swallowed_artifact
            }
        }
        
        add_character_flag = {
            flag = portrait_belly
            days = 3
        }
    }
    
    # Swallow the artifact playfully
    option = {
        name = seduce_vore.1010.a
        
        custom_tooltip = seduce_vore.1010.a.tt
        
        scope:swallowed_artifact = {
            set_variable = {
                name = temporarily_swallowed
                value = yes
                days = 3
            }
        }
        
        hidden_effect = {
            random_list = {
                70 = {
                    modifier = {
                        add = 20
                        scope:target = { has_trait = lustful }
                    }
                    modifier = {
                        add = 15
                        scope:target = { has_trait = gregarious }
                    }
                    modifier = {
                        add = -30
                        scope:target = { has_trait = greedy }
                    }
                    modifier = {
                        add = -25
                        scope:target = { has_trait = wrathful }
                    }
                    modifier = {
                        add = 10
                        scope:target = { 
                            has_relation_lover = root
                        }
                    }
                    
                    seduce_ongoing_add_scheme_modifier_effect = { MILESTONE = 1 OUTCOME = success }
                    
                    scope:target = {
                        add_opinion = {
                            target = root
                            modifier = respect_opinion
                            opinion = 20
                        }
                    }
                    
                    root = {
                        trigger_event = {
                            id = seduce_vore.1011
                            days = 2
                        }
                    }
                }
                30 = {
                    scope:target = {
                        add_opinion = {
                            target = root
                            modifier = disappointed_opinion
                            opinion = -20
                        }
                    }
                    
                    root = {
                        trigger_event = {
                            id = seduce_vore.1012
                            days = 2
                        }
                    }
                }
            }
        }
        
        ai_chance = {
            base = 100
            modifier = {
                add = -50
                scope:target = { has_trait = greedy }
            }
        }
    }
    
    # Don't swallow it (decline the event)
    option = {
        name = seduce_vore.1010.b
        custom_tooltip = seduce_ongoing_decline_event.tt
        
        scope:scheme = {
            set_variable = {
                name = declined_seduce_milestone_1_event
                value = yes
            }
        }
        
        ai_chance = {
            base = 0
            modifier = {
                add = 100
                scope:target = { has_trait = greedy }
            }
        }
    }
    
    after = {
        remove_variable = scheme_event_active_flag
    }
}

# The Game - Owner's perspective (you win or lose)
seduce_vore.1082 = {
    type = character_event
    title = seduce_vore.1080.t
    desc = {
        desc = seduce_vore.1082.opening
        triggered_desc = {
            trigger = { scope:game_type = flag:chess }
            desc = seduce_vore.1082.desc.chess
        }
        triggered_desc = {
            trigger = { scope:game_type = flag:dice }
            desc = seduce_vore.1082.desc.dice
        }
        triggered_desc = {
            trigger = { scope:game_type = flag:riddles }
            desc = seduce_vore.1082.desc.riddles
        }
    }
    theme = seduce_scheme
    
    left_portrait = {
        character = root
        animation = personality_rational
    }
    
    right_portrait = {
        character = scope:target
        animation = thinking
    }
    
    lower_left_portrait = root
    
    trigger = {
        exists = scope:scheme
    }
    
    immediate = {
        # Calculate who wins based on skills
        if = {
            limit = { scope:game_type = flag:chess }
            # Chess - learning skill
            if = {
                limit = {
                    learning >= scope:target.learning
                }
                save_scope_value_as = {
                    name = game_result
                    value = flag:owner_wins
                }
            }
            else = {
                save_scope_value_as = {
                    name = game_result
                    value = flag:target_wins
                }
            }
        }
        else_if = {
            limit = { scope:game_type = flag:dice }
            # Dice - pure luck
            random_list = {
                50 = {
                    save_scope_value_as = {
                        name = game_result
                        value = flag:owner_wins
                    }
                }
                50 = {
                    save_scope_value_as = {
                        name = game_result
                        value = flag:target_wins
                    }
                }
            }
        }
        else_if = {
            limit = { scope:game_type = flag:riddles }
            # Riddles - intrigue skill
            if = {
                limit = {
                    intrigue >= scope:target.intrigue
                }
                save_scope_value_as = {
                    name = game_result
                    value = flag:owner_wins
                }
            }
            else = {
                save_scope_value_as = {
                    name = game_result
                    value = flag:target_wins
                }
            }
        }
        
        # Add belly portrait if you win
        if = {
            limit = { scope:game_result = flag:owner_wins }
            add_character_flag = {
                flag = portrait_belly
                days = 3
            }
        }
    }
    
    # You won! Claim your prize
    option = {
        name = seduce_vore.1082.a
        
        trigger = { scope:game_result = flag:owner_wins }
        
        custom_tooltip = seduce_vore.1082.a.tt
        
        hidden_effect = {
            random_list = {
                80 = {  # Most honor the bet
                    modifier = {
                        add = 15
                        scope:target = { has_trait = honest }
                    }
                    modifier = {
                        add = -30
                        scope:target = { has_trait = deceitful }
                    }
                    
                    # Success: They honor the bet
                    seduce_ongoing_add_scheme_modifier_effect = { MILESTONE = 1 OUTCOME = success }
                    
                    scope:target = {
                        add_opinion = {
                            target = scope:owner
                            modifier = amused_opinion
                            opinion = 15
                        }
                    }
                    
                    # Trigger success event
                    scope:owner = {
                        trigger_event = {
                            id = seduce_vore.1084
                            days = 1
                        }
                    }
                }
                20 = {  # They try to back out
                    # They refuse to honor the bet
                    scope:target = {
                        add_opinion = {
                            target = scope:owner
                            modifier = refusal_opinion
                            opinion = -10
                        }
                    }
                    
                    # Trigger refusal event
                    scope:owner = {
                        trigger_event = {
                            id = seduce_vore.1085
                            days = 1
                        }
                    }
                }
            }
        }
        
        ai_chance = {
            base = 100
        }
    }
    
    # You lost - they get to pick what you do
    option = {
        name = seduce_vore.1082.b
        
        trigger = { scope:game_result = flag:target_wins }
        
        custom_tooltip = seduce_vore.1082.b.tt
        
        # Small prestige hit for losing
        add_prestige = minor_prestige_loss
        
        # They might ask for something in return
        scope:target = {
            trigger_event = {
                id = seduce_vore.1086
                days = 1
            }
        }
        
        ai_chance = {
            base = 100
        }
    }
}
