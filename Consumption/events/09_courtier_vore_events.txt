# 1. The Jester's New Trick
courtier_vore.0001 = {
    type = character_event
    title = courtier_voreTitle.0001
    theme = court
    desc = {
        first_valid = {
            triggered_desc = {
                trigger = { scope:predator = { has_trait = sadistic } }
                desc = courtier_vore.0001.desc.sadistic
            }
            triggered_desc = {
                trigger = { scope:predator = { has_trait = lustful } }
                desc = courtier_vore.0001.desc.lustful
            }
            triggered_desc = {
                trigger = { scope:predator = { has_trait = compassionate } }
                desc = courtier_vore.0001.desc.compassionate
            }
            triggered_desc = {
                trigger = { scope:predator = { has_trait = craven } }
                desc = courtier_vore.0001.desc.craven
            }
            # Fallback generic
            desc = courtier_vore.0001.desc
        }
    }

    cooldown = { years = 3 }

    trigger = {
        court_position:court_jester_court_position = {
            can_vore = yes
        }
        any_courtier_or_guest = {
            count > 0
            can_be_prey = yes
        }
    }

    immediate = {
        court_position:court_jester_court_position = {
            save_scope_as = predator
            add_character_flag = portrait_belly
        }
        # Pick a random guest/courtier target isn't the jester
        random_courtier_or_guest = {
            limit = {
                can_be_prey = yes
                NOT = { this = scope:predator }
            }
            save_scope_as = prey
        }
    }

    left_portrait = {
        character = scope:predator
        animation = schadenfreude
    }
    
    # Showing the Player (Root) watching
    right_portrait = {
        character = root
        animation = intrigue
    }

    # Safe
    option = {
        name = courtier_voreOpt.0001a
        scope:predator = {
            random_orifice_safe_vore_effect = {
                prison_type = vore_dungeon
            }
            stress_impact = {
                compassionate = minor_stress_impact_gain
                sadistic = minor_stress_impact_loss
            }
        }      
        ai_chance = {
            safe_vore_modifier = { PREY = scope:prey }
        }
    }
    
    # Lethal
    option = {
        name = courtier_voreOpt.0001b
        trigger = { global_var:digestion_banned = no }
        ai_chance = {
            lethal_vore_modifier = { PREY = scope:prey }
        }
        scope:predator = {
            random_orifice_lethal_vore_effect = yes
        }
    }

    after = {
        clean_up_effect = yes
    }
}

        character = scope:predator
        animation = seduction

        animation = happiness

        character = scope:prey

                    prison_type = vore_dungeon

            base = 0

        }
    }
    
    cooldown = { years = 3 }

    trigger = {
        any_courtier_or_guest = {
            can_vore = yes

                    has_relation_rival = root 
                    has_relation_nemesis = root
                }
            }

            }
        }
        any_courtier_or_guest = {
            OR = {
                has_relation_rival = root
                has_relation_nemesis = root
                opinion = {
                    target = root
                    value < -60
                }
            }
            can_be_prey = yes
        }
    }

    immediate = {
        random_courtier_or_guest = {
            limit = {
                can_vore = yes
                NOT = { 
                    OR = {
                        has_relation_rival = root 
                        has_relation_nemesis = root
                    }
                }
                opinion = {
                    target = root
                    value > 20
                }
            }
            save_scope_as = predator
        }
        random_courtier_or_guest = {
            limit = {
                OR = {
                    has_relation_rival = root
                    has_relation_nemesis = root
                    opinion = {
                        target = root
                        value < -60
                    }
                }
                can_be_prey = yes
            }
            save_scope_as = prey
        }
        scope:predator = {
            add_character_flag = portrait_belly
        }
    }

    left_portrait = {
        character = scope:predator
        animation = smile
    }
    
    right_portrait = {
        character = root
        animation = schadenfreude
    }
    
    lower_left_portrait = {
        character = scope:prey
    }

    # Lethal - Dispose of rival
    option = {
        name = courtier_voreOpt.0030a
        
        trigger = {
            global_var:digestion_banned = no
        }

        ai_chance = {
            base = 0
            lethal_vore_modifier = { PREY = scope:prey}
        }
        
        scope:predator = {
            if = {
                limit = { global_var:anal_banned = no }
                lethal_vore_effect = {
                    type = anal
                }
            }
            else = {
                random_orifice_lethal_vore_effect = yes

                target = root
                modifier = grateful_opinion
                opinion = 25
            }

        stress_impact = {
            compassionate = major_stress_impact_gain
            sadistic = minor_stress_impact_loss
            vengeful = minor_stress_impact_loss
        }
        
    }

    # Safe - Imprison rival

            base = 20
            modifier = {
                add = 40
                has_trait = just
            }
            modifier = {
                add = 30
                has_trait = compassionate
            }
        }
    }
    
    after = {
        clean_up_effect = yes
    }
}

# Event: Concubine Devours Concubine
courtier_vore.0050 = {
    type = character_event
    title = courtier_voreTitle.0050
    desc = {
        first_valid = {
            triggered_desc = {
                trigger = { scope:predator = { has_trait = sadistic } }

            }
            triggered_desc = {
                trigger = { scope:predator = { has_trait = lustful } }
                desc = courtier_vore.0050.desc.lustful
            }
            triggered_desc = {
                trigger = { scope:predator = { has_trait = gregarious } }
                desc = courtier_vore.0050.desc.gregarious
            }
            triggered_desc = {
                trigger = { scope:predator = { has_trait = compassionate } }
                desc = courtier_vore.0050.desc.compassionate
            }
            triggered_desc = {
                trigger = { scope:predator = { has_trait = craven } }
                desc = courtier_vore.0050.desc.craven
            }
            triggered_desc = {
                trigger = { scope:predator = { has_trait = eccentric } }
                desc = courtier_vore.0050.desc.envy
            }
            triggered_desc = {
                trigger = { scope:predator = { has_trait = gluttonous } }
                desc = courtier_vore.0050.desc.gluttony
            }
            desc = courtier_vore.0050.desc.generic
        }
    }
    theme = intrigue
    
    left_portrait = {
        character = root
        animation = idle_standing
    }
    center_portrait = {
        character = scope:predator
        animation = schadenfreude
    }
    lower_right_portrait = {
        character = scope:prey
        animation = fear
    }

    trigger = {      
        any_concubine = { 
            can_vore = yes 
        }
        
        any_concubine = { 
            count >= 2
            can_be_prey = yes 
        }
    }

    cooldown = { years = 3 }

    immediate = {
        random_concubine = {
            limit = { can_vore = yes }
            save_scope_as = predator
        }
        
        ordered_concubine = {
            limit = { 
                can_be_prey = yes
                NOT = { this = scope:predator }
            }
            order_by = {
                value = "opinion(scope:predator)"
                multiply = -1
            }
            save_scope_as = prey
        }
        
        scope:predator = {
            add_character_flag = portrait_belly
        }
    }

    # Option 1: Punish the Predator (Stop the fight)
    option = {
        name = courtier_vore.0050.opt.stop
        
        hidden_effect = {
            scope:predator = {
                add_stress = medium_stress_gain
                add_opinion = {
                    target = root
                    modifier = angry_opinion
                    opinion = -20
                }
                remove_character_flag = portrait_belly
            }
            
            scope:prey = {
                add_opinion = {
                    target = root
                    modifier = vored_me_unwilling_opinion
                    opinion = -15
                }
            }
            
            stress_impact = {
                just = minor_stress_impact_loss
                compassionate = minor_stress_impact_loss
                callous = minor_stress_impact_gain
            }
        }
    }
    
    # Option 2: Allow the Feast (Lethal)
    option = {
        name = courtier_vore.0050.opt.lethal
    
        trigger = { global_var:digestion_banned = no }
        
        scope:predator = {
            if = {
                limit = { global_var:oral_banned = no }
                lethal_vore_effect = { type = oral }
            } 
            else = {
                random_orifice_lethal_vore_effect = yes
            }
            add_opinion = {
                target = root
                modifier = grateful_opinion
                opinion = 25
            }
        }
        
        stress_impact = {
            compassionate = major_stress_impact_gain
            sadistic = minor_stress_impact_loss
            callous = minor_stress_impact_loss
        }
        
        ai_chance = {
            base = 0
            lethal_vore_modifier = { PREY = scope:prey }
        }
    }
    
    # Option 3: Safe Storage (Non-Lethal)
    option = {
        name = courtier_vore.0050.opt.safe
        
        scope:predator = {
            if = {
                limit = {
                    is_female = yes
                    global_var:unbirth_banned = no
                }
                safe_vore_effect = {
                    type = unbirth
                    prison_type = vore_dungeon
                }
            }
            else_if = {
                limit = {
                    is_male = yes
                    global_var:cock_banned = no
                }
                safe_vore_effect = {
                    type = cock
                    prison_type = vore_dungeon
                }
            }
            else = {
                random_orifice_safe_vore_effect = {
                    prison_type = vore_dungeon
                }
            }
            add_opinion = {
                target = root
                modifier = grateful_opinion
                opinion = 20
            }
        }
        
        stress_impact = {
            compassionate = minor_stress_impact_loss
        }
        
        ai_chance = {
            base = 0
            safe_vore_modifier = { PREY = scope:prey }
        }
    }
}
