# Event: A hunting party member is ambushed by the beast (The Setup)
hunt_vore.0001 = {
    type = character_event
    title = hunt.9000.t
    desc = hunt.9000.desc
    theme = hunting

    trigger = {
        global_var:feral_banned = no
    }
    
    # NOTE: scope:victim is set in immediate below.
    # Portrait will fall back gracefully on first fire since immediate runs after portrait evaluation.
    # To fix properly, victim selection would need to happen earlier in the call chain.
    left_portrait = {
        character = scope:victim
        animation = fear
    }
    
    immediate = {
        # Pick a random guest/party member to be the victim
        # FIX: Added scope:activity guard - verify this is passed from on_hunt_random_pulse context
        scope:activity = {
            random_attending_character = {
                limit = { 
                    is_alive = yes 
                    is_imprisoned = no 
                    NOT = { this = root } 
                }
                save_scope_as = victim
            }
        }
    }

    option = {
        name = hunt.9000.opt.a
        trigger_event = hunt_vore.0002
    }
}

# Oral Vore (Feral)
hunt_vore.0003 = {
    type = character_event
    title = hunt.0003.t
    theme = hunting
    
    desc = {
        first_valid = {
            triggered_desc = {
                trigger = {
                    OR = {
                        scope:activity.var:animal_type = flag:tiger
                        scope:activity.var:animal_type = flag:leopard
                        scope:activity.var:animal_type = flag:lion
                        scope:activity.var:animal_type = flag:bear
                    }
                }
                desc = hunt_vore.0003.desc.claws
            }
            triggered_desc = {
                trigger = {
                    OR = {
                        scope:activity.var:animal_type = flag:hyena
                        scope:activity.var:animal_type = flag:wolf
                        scope:activity.var:animal_type = flag:lynx
                    }
                }
                desc = hunt_vore.0003.desc.fangs
            }
            desc = hunt_vore.0003.desc.fallback
        }
    }

    right_portrait = {
        character = scope:victim
        animation = fear
    }

    option = {
        name = hunt.0003.opt.a
        hidden_effect = {
            scope:victim = {
                if = {
                    limit = { global_var:digestion_banned = no }
                    death = { death_reason = death_digested_by_wild_beast }
                }
                else = {
                    death = {death_reason = death_vored_by_wild_beast }
                }
            }
            root = {
                add_stress = 10
                add_prestige = -5
            }
        }
    }
}

# Event: Successful Vore Outcome (Hunt) - Player/character eats someone during the hunt
hunt_vore.0007 = {
    type = activity_event
    title = hunt_vore.0003.t
    desc = {
        first_valid = {
            # ORAL
            triggered_desc = {
                trigger = { has_trait = oral_predator }
                desc = {
                    random_valid = {
                        desc = hunt_vore.0003.desc.oral.1
                        desc = hunt_vore.0003.desc.oral.2
                    }
                }
            }
            # ANAL
            triggered_desc = {
                trigger = { has_trait = anal_predator }
                desc = {
                    random_valid = {
                        desc = hunt_vore.0003.desc.anal.1
                        desc = hunt_vore.0003.desc.anal.2
                    }
                }
            }
            # UNBIRTH
            triggered_desc = {
                trigger = { has_trait = unbirth_predator }
                desc = {
                    random_valid = {
                        desc = hunt_vore.0003.desc.unbirth.1
                        desc = hunt_vore.0003.desc.unbirth.2
                    }
                }
            }
            # COCK
            triggered_desc = {
                trigger = { has_trait = cock_predator }
                desc = {
                    random_valid = {
                        desc = hunt_vore.0003.desc.cock.1
                        desc = hunt_vore.0003.desc.cock.2
                    }
                }
            }
        }
    }
    theme = hunting
    
    left_portrait = {
        character = root
        animation = schadenfreude
    }
    
    center_portrait = {
        character = scope:witness
        triggered_animation = {
            trigger = { has_trait = lustful }
            animation = intrigue
        }
        triggered_animation = {
            trigger = { has_trait = compassionate }
            animation = worry
        }
        triggered_animation = {
            trigger = {
                NOT = { has_trait = lustful }
                NOT = { has_trait = compassionate }
            }
            animation = shock
        }
    }
    
    right_portrait = {
        character = scope:prey
        animation = fear
    }
    
    override_background = { reference = wilderness }

    immediate = {
        scope:victim = { save_scope_as = prey }

        # Setup Witness Scope (Random participant who isn't the prey)
        scope:activity = {
            random_attending_character = {
                limit = { 
                    is_alive = yes
                    NOT = { this = scope:prey }
                }
                save_scope_as = witness
            }
        }
        
        # Execute Vore Logic (Imprisonment)
        random_orifice_safe_vore_effect = { 
            prison_type = vore_dungeon
        }
        
        # Add Activity Log
        scope:activity = {
            add_activity_log_entry = {
                key = hunt_vore_success_log
                score = 25
                tags = { random bad }
                character = root
                target = scope:prey
            }
        }
    }

    # Option 1: Continue Hunt
    option = {
        name = hunt_vore.0003.opt.continue
        trigger = {
            is_alive = yes
        }
        flavor = hunt_vore.0003.flavor.continue
        
        # FIX: Moved root effects outside scope:activity block
        scope:activity = {
            add_activity_log_entry = {
                key = hunt_vore_continue_log
                score = 25
                tags = { random bad }
                character = root
                target = scope:prey
            }
        }
        root = {
            add_stress = 10
            add_prestige = 5
        }
        
        ai_chance = {
            base = 50
            ai_value_modifier = {
                ai_energy = 1
                ai_rationality = -0.5
            }
        }
    }

    # Option 2: End Hunt (Return Home)
    option = {
        name = hunt_vore.0003.opt.end
        flavor = hunt_vore.0003.flavor.end
        
        # FIX: Moved root effects outside scope:activity block
        scope:activity = {
            set_variable = {
                name = hunt_invalidated
            }
            set_variable = {
                name = hunt_invalidated_flavor
                value = root
            }
            add_activity_log_entry = {
                key = hunt_vore_end_log
                score = 25
                tags = { random bad }
                character = root
                target = scope:prey
            }
        }
        root = {
            custom_tooltip = hunt_voreTooltip.0003b
            # stress_impact = {
            #     sadistic = minor_stress_relief_gain
            #     gluttonous = minor_stress_relief_gain
            #     chaste = major_stress_impact_gain
            # }
        }
        
        ai_chance = {
            base = 50
            ai_value_modifier = {
                ai_energy = -0.5
                ai_rationality = 1
            }
        }
    }
}
