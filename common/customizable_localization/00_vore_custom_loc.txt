# ==========================================
# CUSTOM LOCALIZATION FOR VORE
# ==========================================

# 1. Prison Location Type (Existing)
# This changes what the prison building/area is called.
PrisonType = {
    type = character
    text = {
        trigger = { has_character_flag = vored }
        localization_key = prisontype_stomach
	}
}

# 2. Role: Prisoner -> Prey
# This changes the character's role label when hovering over them or in certain menus.
prisoner = {
    type = character
    text = {
        trigger = { has_character_flag = vored }
        localization_key = concept_prey_vore
    }
}

# 3. Concept: Dungeon -> Stomach
# This changes the text when referring to the dungeon location.
dungeon = {
    type = character
    text = {
        trigger = {
            AND = {
                has_character_flag = vore_predator
            }
        }
        localization_key = concept_dungeon_stomach
    }
}

# 4. Status: Imprisonment -> Digestion/Storage
# This changes the status icon tooltip.
imprisonment = {
    type = character
    text = {
        trigger = { has_character_flag = vored }
        localization_key = concept_imprisonment_storage
    }
}
